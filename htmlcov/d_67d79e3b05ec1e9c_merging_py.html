39def fast_agglomerate(data, splist, radius, method='distance', scale=1.5): 

40 """ 

41 Implement CLASSIX's merging with disjoint-set data structure, default choice for the merging. 

42  

43 Parameters 

44 ---------- 

45 data : numpy.ndarray 

46 The input that is array-like of shape (n_samples,). 

47  

48 splist : numpy.ndarray 

49 List of starting points formed in the aggregation. 

50 

51 radius : float 

52 The tolerance to control the aggregation. If the distance between the starting point  

53 of a group and another data point is less than or equal to the tolerance, 

54 the point is allocated to that group. For details, we refer users to [1]. 

55 

56 method : str, default='distance' 

57 Method for group merging, Available options are: 

58  

59 - 'density': two groups are merged if the density of data points in their intersection  

60 is at least as high the smaller density of both groups. This option uses the disjoint  

61 set structure to speedup agglomerate. 

62  

63 - 'distance': two groups are merged if the distance of their starting points is at  

64 most scale*radius (the parameter above). This option uses the disjoint  

65 set structure to speedup agglomerate. 

66  

67 scale : float 

68 Design for distance-clustering, when distance between the two starting points  

69 associated with two distinct groups smaller than scale*radius, then the two groups merge. 

70 

71 Returns 

72 ------- 

73 labels_set : list 

74 Connected components of graph with groups as vertices. 

75  

76 connected_pairs_store : list 

77 List for connected group labels. 

78 

79 References 

80 ---------- 

81 [1] X. Chen and S. Güttel. Fast and explainable sorted based clustering, 2022 

82 

83 """

86 connected_pairs_store = [] 

87 if method == "density":

112 den1 = np.count_nonzero(c1) / volume 

113 # den1 = splist[int(i), 2] / volume # density(splist[int(i), 2], volume = volume)  

114 for j in select_stps: 

115 sp2 = splist[int(j), 3:] 

116 

117 c2 = np.linalg.norm(data-sp2, ord=2, axis=-1) <= radius 

118 den2 = np.count_nonzero(c2) / volume

205def findParent(s): 

206 """Find parent of node.""" 

207 if (s.data != s.parent.data) : 

208 s.parent = findParent((s.parent)) 

209 return s.parent

213def merge(s1, s2): 

214 """Merge the roots of two node.""" 

215 parent_of_s1 = findParent(s1) 

216 parent_of_s2 = findParent(s2) 

217 

218 if (parent_of_s1.data != parent_of_s2.data) :

280def check_if_overlap(starting_point, spo, radius, scale = 1): 

281 """Check if two groups formed by aggregation overlap 

282 """ 

283 return np.linalg.norm(starting_point - spo, ord=2, axis=-1) <= 2 * scale * radius

287def cal_inter_density(starting_point, spo, radius, num): 

288 """Calculate the density of intersection (lens) 

289 """ 

290 in_volume = cal_inter_volume(starting_point, spo, radius) 

291 return num / in_volume

305def cal_inter_volume(starting_point, spo, radius): 

306 """ 

307 Returns the volume of the intersection of two spheres in n-dimensional space. 

308 The radius of the two spheres is r and the distance of their centers is d. 

309 For d=0 the function returns the volume of full sphere. 

310 Reference: https://math.stackexchange.com/questions/162250/how-to-compute-the-volume-of-intersection-between-two-hyperspheres 

311 

312 """ 

313 

314 dim =starting_point.shape[0] 

315 dist = np.linalg.norm(starting_point - spo, ord=2, axis=-1) # the distance between the two groups 

316 

317 if dist > 2*radius: 

318 return 0 

319 c = dist / 2 

320 return np.pi**(dim/2)/gamma(dim/2 + 1)*(radius**dim)*betainc((dim + 1)/2, 1/2, 1 - c**2/radius**2)
